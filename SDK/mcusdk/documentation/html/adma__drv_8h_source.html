109 typedef union {

111  hwint32 buffer_size_low_part : 15; /* 0:14 sw=RW hw=RO 0x0, B0 increase the buffer size to 18bit, another 3 bit is in other register*/

113  hwint32 s_cnt : 7; /* 18:24 sw=RW hw=RO 0x0 */

117  hwint32 block_interrupt : 1; /* 30:30 sw=RW hw=RO 0x0 */
