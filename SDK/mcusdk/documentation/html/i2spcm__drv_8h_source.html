112 {

124 #define I2SPCM_EN_NO_FLAG 0 //means don't do reset, and i2s channels will be enabled

126 #define I2SPCM_EN_FLAG_WITH_RST_TX 2 //when this bit is set, means reset tx
127 #define I2SPCM_EN_FLAG_WITHOUT_EN 4 //when this bit is set, means don't enable i2s rx/tx channels, but it will support the i2s fifo full mechanism, when fifo full, adma will bloacked
